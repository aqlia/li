fun main(args: Array<String>) {
    val secret = (1..100).random()
    var attempts = 0
    println("Guess a number between 1 and 100")
    var guess = readLine()!!.toInt()
    if (guess == secret) {
        println("You guessed the number in 1 attempt!")
        return
    }
    while (guess != secret) {
        if (guess < secret) println("Too low")
        else if (guess > secret) println("Too high")
        else break
        attempts++
        guess = readLine()!!.toInt()
    }
    println("You guessed the number in ${attempts+2} attempts!")
}
